<script lang="ts">
import { defineComponent, ref } from 'vue';
 
export default defineComponent({
	props: {
		phoneNumber: {
			type: String,
			default: "+372 5289 6572",
		},
		email: {
			type: String,
			default: "anna.tamm@gmail.com",
		},
		customerId: {
			type: String,
			default: "38912052254",
		}
	},
	data() {
		return {
			cellPhone: this.phoneNumber,
			emailAddress: this.email,
			isEditing: false,
			isPhoneNumberValid: true,
			isEmailValid: true,
			errorMessage: 'Incorrect contact details',
		};
	},
	methods: {
		submitForm() {
			if (this.isPhoneNumberValid && this.isEmailValid) {
				this.isEditing = false;
			}
			else {
				this.errorMessage = "Incorrect contact details";
			}
		},
		validateEmail() {
			const emailRegex = /^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,4}$/;
			this.isEmailValid = emailRegex.test(this.emailAddress);
			this.errorMessage = !this.isEmailValid ? "Incorrect email address" : "";
		},
		validatePhoneNumber() {
			const formattedNumber = this.cellPhone.split(" ").join("");
			const phoneNumberRegex = /^(?:\+372|00372|8)[1-9]\d{6,7}$/;
			this.isPhoneNumberValid = phoneNumberRegex.test(formattedNumber) && formattedNumber.length === 12;
			this.errorMessage = !this.isPhoneNumberValid ? "Incorrect phone number" : "";
		}
	},
});
</script>
